<script setup>
import { ref, onMounted } from 'vue'
import { Dropzone } from "@deltablot/dropzone"


const dropzoneRef = ref()

onMounted(() => {
    console.log(dropzoneRef.value)
    const myDropzone = new Dropzone(dropzoneRef.value, {
    autoProcessQueue: false,
    paramName: 'file',
    chunking: true,
    forceChunking: true,
    url: '/api/v1/analysers',
    maxFiles: 1,
    maxFilesize: 5 * 1024, // in MiB
    chunkSize: 10 * 1024 * 1024 // in B
  })})


</script>

<template>
    <div ref="dropzoneRef" class="dropzone"></div>
</template>
